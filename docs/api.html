Header Required Value
x-api-key Yes <your hybridClientApiKey>
Content-Type Yes application/json

GET /v1/points/:externalUserId #

GET /v1/points/:externalUserId

Called by Reward Geni after SSO login to fetch the user's current point balance grouped by blocks.

Code Examples

cURL
curl -X GET "https://your-api.example.com/v1/points/EMP001" \
    -H "x-api-key: sk_live_your_api_key"
JavaScript
const response = await fetch('https://your-api.example.com/v1/points/EMP001', {
    method: 'GET',
    headers: {
        'x-api-key': 'sk_live_your_api_key'
    }
});

const data = await response.json();
console.log(data);
Node.js
const axios = require('axios');

const { data } = await axios.get(
    'https://your-api.example.com/v1/points/EMP001',
    {
        headers: {
            'x-api-key': 'sk_live_your_api_key'
        }
    }
);

console.log(data);

  "externalUserId": "EMP001",

    {
      "blockId": "BLOCK_001",

    },

POST /v1/redeem #

POST /v1/redeem

Called before order creation to deduct points from the user.

Request Body

JSON
{
  "externalUserId": "EMP001",
  "points": 500,
  "orderId": "ORD-20250101-ABC123",
  "orderNumber": "ORD-20250101-ABC123",
  "orderType": "PHYSICAL",
  "orderDate": "2025-01-01T10:00:00.000Z",
  "orderItem": {
    "id": "item-uuid-here",
    "orderType": "PHYSICAL",
    "quantity": 2,
    "product": {
      "productId": "prod-uuid-here",
      "productName": "Premium Wireless Headphones"
    }
  },
  "description": "Redeem 500 points for order ORD-20250101-ABC123"
}

Code Examples

cURL
curl -X POST "https://your-api.example.com/v1/redeem" \
    -H "Content-Type: application/json" \
    -H "x-api-key: sk_live_your_api_key" \
    -d '{
        "externalUserId": "EMP001",
        "points": 500,
        "orderId": "ORD-20250101-ABC123",
        "orderNumber": "ORD-20250101-ABC123",
        "orderType": "PHYSICAL",
        "orderDate": "2025-01-01T10:00:00.000Z",
        "orderItem": {
            "id": "item-uuid-here",
            "orderType": "PHYSICAL",
            "quantity": 2,
            "product": {
                "productId": "prod-uuid-here",
                "productName": "Premium Wireless Headphones"
            }
        },
        "description": "Redeem 500 points for order ORD-20250101-ABC123"
    }'
JavaScript
const response = await fetch('https://your-api.example.com/v1/redeem', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
        'x-api-key': 'sk_live_your_api_key'
    },
    body: JSON.stringify({
        externalUserId: 'EMP001',
        points: 500,
        orderId: 'ORD-20250101-ABC123',
        orderNumber: 'ORD-20250101-ABC123',
        orderType: 'PHYSICAL',
        orderDate: '2025-01-01T10:00:00.000Z',
        orderItem: {
            id: 'item-uuid-here',
            orderType: 'PHYSICAL',
            quantity: 2,
            product: {
                productId: 'prod-uuid-here',
                productName: 'Premium Wireless Headphones'
            }
        },
        description: 'Redeem 500 points for order ORD-20250101-ABC123'
    })
});

const data = await response.json();
console.log(data);
Node.js
const axios = require('axios');

const { data } = await axios.post(
    'https://your-api.example.com/v1/redeem',
    {
        externalUserId: 'EMP001',
        points: 500,
        orderId: 'ORD-20250101-ABC123',
        orderNumber: 'ORD-20250101-ABC123',
        orderType: 'PHYSICAL',
        orderDate: '2025-01-01T10:00:00.000Z',
        orderItem: {
            id: 'item-uuid-here',
            orderType: 'PHYSICAL',
            quantity: 2,
            product: {
                productId: 'prod-uuid-here',
                productName: 'Premium Wireless Headphones'
            }
        },
        description: 'Redeem 500 points for order ORD-20250101-ABC123'
    },
    {
        headers: {
            'Content-Type': 'application/json',
            'x-api-key': 'sk_live_your_api_key'
        }
    }
);

console.log(data);

Success Response (200)

JSON
{
  "success": true,
  "transactionId": "TXN_CLIENT_12345",
  "pointsRedeemed": 500,
  "remainingPoints": 1000
}

Error Responses

402
{

  "code": "HC_402_001",
  "message": "Insufficient points for redemption",
  "availablePoints": 300
}
404
{
  "success": false,
  "code": "HC_404_001",
  "message": "User not found"
}
500
{
  "success": false,
  "code": "HC_500_001",
  "message": "Internal server error"
}
503
{
  "success": false,
  "code": "HC_503_001",
  "message": "Redemption service temporarily unavailable. Please try again later."
}
